**2. AccountingPeriodServiceImpl:**

```java
@Service
@Transactional
public class AccountingPeriodServiceImpl implements AccountingPeriodService {

    public int seedForTenant(UUID tenantId, int fiscalYear) {
        // Pakistan FY: period 1 = July (month 7) of (fiscalYear-1), period 12 = June (month 6) of fiscalYear
        int seeded = 0;
        int startCalYear = fiscalYear - 1;
        for (int periodNo = 1; periodNo <= 12; periodNo++) {
            int month = ((6 + periodNo - 1) % 12) + 1; // maps 1→7, 2→8, ..., 6→12, 7→1, ..., 12→6
            int year = (periodNo <= 6) ? startCalYear : fiscalYear;
            LocalDate start = LocalDate.of(year, month, 1);
            LocalDate end = start.withDayOfMonth(start.lengthOfMonth());

            // Idempotent: skip if already exists
            if (!periodRepo.existsByFiscalYearAndPeriodNo(fiscalYear, periodNo)) {
                AccountingPeriod p = new AccountingPeriod();
                p.setFiscalYear(fiscalYear);
                p.setPeriodNo(periodNo);
                p.setStartDate(start);
                p.setEndDate(end);
                p.setStatus(PeriodStatus.OPEN);
                periodRepo.save(p); // tenant_id set by TenantAuditableEntity via TenantContext
                seeded++;
            }
        }
        return seeded;
    }

    @Transactional(readOnly = true)
    public Optional<AccountingPeriod> getCurrentOpenPeriod() {
        return periodRepo.findFirstByStatusOrderByEndDateDesc(PeriodStatus.OPEN);
    }
}
```

```java
@Service
@Transactional
public class PeriodCloseService {

    public AccountingPeriodDto close(UUID periodId, boolean totpVerified) {
        // AUTH-07: finance.period.close requires TOTP
        if (!totpVerified) {
            throw new TotpRequiredException();  // → 403 TOTP_REQUIRED
        }

        AccountingPeriod period = periodRepo.findById(periodId)
            .orElseThrow(() -> new PeriodNotFoundException(periodId));

        if (period.getStatus() == PeriodStatus.LOCKED) {
            throw new PeriodAlreadyLockedException(periodId);  // → 409
        }

        // Pre-close checks via internal Feign (Phase 6: stubs return 0)
        long openOrders = posClient.getOpenOrderCount(
            period.getStartDate(), period.getEndDate());  // TODO: Phase 7 implement real endpoint
        long pendingGrn = inventoryClient.getPendingGrnCount(period.getEndDate());  // TODO: Phase 8
        long unmatchedInvoices = purchasingClient.getUnmatchedInvoiceCount(period.getEndDate());  // TODO: Phase 10

        if (openOrders > 0) throw new PeriodPreCheckException("open orders: " + openOrders);
        if (pendingGrn > 0) throw new PeriodPreCheckException("pending GRNs: " + pendingGrn);
        if (unmatchedInvoices > 0) throw new PeriodPreCheckException("unmatched invoices: " + unmatchedInvoices);

        period.setStatus(PeriodStatus.LOCKED);
        period.setLockedBy(TenantContext.getCurrentUserId());
        period.setLockedAt(Instant.now());
        periodRepo.save(period);

        return periodMapper.toDto(period);
    }
}
```
