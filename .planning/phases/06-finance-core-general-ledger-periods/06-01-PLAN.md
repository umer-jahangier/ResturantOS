- Tables: `chart_of_accounts`, `accounting_periods`, `je_sequences`, `journal_entries`, `journal_lines`
- On `chart_of_accounts`: `UNIQUE(tenant_id, code)`; on `accounting_periods`: `UNIQUE(tenant_id, fiscal_year, period_no)`, CHECK `period_no BETWEEN 1 AND 12`, CHECK `status IN ('OPEN','LOCKED')`.
- RLS on all 4 tables: `ALTER TABLE x ENABLE ROW LEVEL SECURITY; CREATE POLICY tenant_isolation ON x USING (tenant_id = current_setting('app.current_tenant_id', TRUE)::UUID);` — use TRUE (permissive) not FALSE.
- Deferred balance trigger:
  ```sql
  CREATE OR REPLACE FUNCTION check_je_balance() RETURNS trigger AS $$
  DECLARE v_debit BIGINT; v_credit BIGINT;
  BEGIN
    SELECT COALESCE(SUM(debit_paisa),0), COALESCE(SUM(credit_paisa),0)
    INTO v_debit, v_credit FROM journal_lines WHERE je_id = NEW.je_id;
    IF v_debit <> v_credit THEN
      RAISE EXCEPTION 'Journal entry % is not balanced (DR=% CR=%)', NEW.je_id, v_debit, v_credit;
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;

  CREATE CONSTRAINT TRIGGER trg_je_balance
    AFTER INSERT OR UPDATE ON journal_lines
    DEFERRABLE INITIALLY DEFERRED
    FOR EACH ROW EXECUTE FUNCTION check_je_balance();
  ```
- Immutability triggers:
  ```sql
  CREATE OR REPLACE FUNCTION protect_posted_je() RETURNS trigger AS $$
  BEGIN
    IF OLD.status = 'POSTED' THEN
      -- Allow ONLY the reversed_by_je link update (no other field changes)
      IF TG_OP = 'UPDATE'
         AND OLD.reversed_by_je IS NULL
         AND NEW.reversed_by_je IS NOT NULL
         AND OLD.status = NEW.status THEN
        RETURN NEW;
      END IF;
      RAISE EXCEPTION 'Journal entry % is POSTED and immutable.', OLD.id;
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;
  CREATE TRIGGER trg_je_immutable BEFORE UPDATE OR DELETE ON journal_entries FOR EACH ROW EXECUTE FUNCTION protect_posted_je();

  CREATE OR REPLACE FUNCTION protect_posted_je_line() RETURNS trigger AS $$
  DECLARE v_status TEXT;
  BEGIN
    SELECT status INTO v_status FROM journal_entries WHERE id = OLD.je_id;
    IF v_status = 'POSTED' THEN
      RAISE EXCEPTION 'Cannot modify lines of POSTED JE %.', OLD.je_id;
    END IF;
    RETURN NEW;
  END;
  $$ LANGUAGE plpgsql;
  CREATE TRIGGER trg_je_line_immutable BEFORE UPDATE OR DELETE ON journal_lines FOR EACH ROW EXECUTE FUNCTION protect_posted_je_line();
  ```

```java
        // Validate lines present, branchId provided
        // Resolve period by entryDate (find the accounting period containing entryDate)
        AccountingPeriod period = findPeriodForDate(req.entryDate());
        if (period.getStatus() == PeriodStatus.LOCKED) throw new PeriodLockedException(period.getId());

        JournalEntry je = new JournalEntry();
        // Never set je.setId() before save — Spring Data calls merge() if ID is non-null [03-02-B]
        je.setPeriod(period);
        je.setEntryDate(req.entryDate());
        je.setDescription(req.description());
        je.setBranchId(req.branchId());
        je.setStatus(JeStatus.DRAFT);
        je.setSourceType(req.sourceType());
        je.setSourceId(req.sourceId());

        for (var lineReq : req.lines()) {
            JournalLine line = new JournalLine();
            line.setAccountCode(lineReq.accountCode());
            line.setDebitPaisa(lineReq.debitPaisa());
            line.setCreditPaisa(lineReq.creditPaisa());
            line.setDescription(lineReq.description());
            je.getLines().add(line);
            line.setJournalEntry(je);
        }
        jeRepo.save(je); // lines saved via cascade, deferred trigger registered
        // NOTE: trigger fires at commit (end of @Transactional), NOT here
        return mapper.toDto(je);
    }

    public JournalEntryDto post(UUID jeId) {
        JournalEntry je = jeRepo.findById(jeId).orElseThrow(() -> new JeNotFoundException(jeId));
        if (je.getStatus() != JeStatus.DRAFT) throw new JeAlreadyPostedException(jeId);
        AccountingPeriod period = je.getPeriod();
        if (period.getStatus() == PeriodStatus.LOCKED) throw new PeriodLockedException(period.getId());

        // Assign human-readable entry number
        int seq = jeSeqRepo.incrementAndGet(TenantContext.getCurrentTenantId(), period.getFiscalYear());
        je.setEntryNo("JE-" + period.getFiscalYear() + "-" + String.format("%06d", seq));
        je.setStatus(JeStatus.POSTED);
        je.setPostedBy(TenantContext.getCurrentUserId());

        jeRepo.save(je);
        // commit → trg_je_balance fires → DataIntegrityViolationException if unbalanced
        return mapper.toDto(je);
    }

    public JournalEntryDto reverse(UUID jeId) {
        JournalEntry orig = jeRepo.findById(jeId).orElseThrow(() -> new JeNotFoundException(jeId));
        if (orig.getStatus() != JeStatus.POSTED) throw new IllegalStateException("Only POSTED JEs can be reversed");
        if (orig.getReversedByJe() != null) throw new IllegalStateException("JE already reversed");

        AccountingPeriod period = orig.getPeriod();
        if (period.getStatus() == PeriodStatus.LOCKED) throw new PeriodLockedException(period.getId());

        JournalEntry rev = new JournalEntry();
        rev.setPeriod(period);
        rev.setEntryDate(LocalDate.now());
        rev.setDescription("Reversal of " + orig.getEntryNo());
        rev.setBranchId(orig.getBranchId());
        rev.setReversal(true);
        rev.setReversalOfJe(orig.getId());
        rev.setStatus(JeStatus.DRAFT); // will be posted below

        for (JournalLine origLine : orig.getLines()) {
            JournalLine revLine = new JournalLine();
            revLine.setAccountCode(origLine.getAccountCode());
            // Swap debit/credit for reversal
            revLine.setDebitPaisa(origLine.getCreditPaisa());
            revLine.setCreditPaisa(origLine.getDebitPaisa());
            revLine.setDescription("Reversal: " + origLine.getDescription());
            rev.getLines().add(revLine);
            revLine.setJournalEntry(rev);
        }
        jeRepo.save(rev);

        // Post the reversal (balance trigger will fire at commit)
        int seq = jeSeqRepo.incrementAndGet(TenantContext.getCurrentTenantId(), period.getFiscalYear());
        rev.setEntryNo("REV-" + orig.getEntryNo().substring(3)); // REV-2026-000001
        rev.setStatus(JeStatus.POSTED);
        rev.setPostedBy(TenantContext.getCurrentUserId());
        jeRepo.save(rev);

        // Link original JE to reversal — the ONLY allowed UPDATE on a POSTED JE
        // The immutability trigger exempts: OLD.reversed_by_je IS NULL AND NEW.reversed_by_je IS NOT NULL AND status unchanged
        orig.setReversedByJe(rev.getId());
        jeRepo.save(orig);

        return mapper.toDto(rev);
    }
}
```
**CRITICAL pitfall:** `post()` MUST be annotated `@Transactional`. The deferred trigger fires at the point the database connection commits, which is when Spring commits the `@Transactional` method. If `post()` is NOT `@Transactional`, the test will never see the trigger fire.

```java
@ExceptionHandler(DataIntegrityViolationException.class)
public ResponseEntity<ApiError> handleUnbalanced(DataIntegrityViolationException ex) {
    if (ex.getMessage() != null && ex.getMessage().contains("not balanced")) {
        return ResponseEntity.unprocessableEntity()
            .body(ApiError.of("JE_UNBALANCED", "Journal entry is not balanced"));
    }

        .body(ApiError.of("PERIOD_LOCKED", "Period " + ex.getPeriodId() + " is locked"));
}
```
